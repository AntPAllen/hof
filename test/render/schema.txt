exec hof render data.yaml schema.cue -T 'template.txt:vals@schemas.vals'
cmp stdout expected.txt

-- schema.cue --
schemas: {
	vals: [...{ name: string, data: int }]
	#foo: { bar: string }
}
-- data.yaml --
vals:
  - name: "a"
    data: 1
  - name: "b"
    data: 2

-- template.txt --
{{ range . -}}
{{ .name }} is {{ .data }}
{{ end }}
-- expected.txt --
a is 1
b is 2
